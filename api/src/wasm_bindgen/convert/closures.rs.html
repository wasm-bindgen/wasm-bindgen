19macro_rules! closures {
20    // Unwind safe passing
21    ([$($maybe_unwind_safe:tt)*] $($rest:tt)*) => {
22        closures!(@process [$($maybe_unwind_safe)*] $($rest)*);
23    };
24
25    // One-arity recurse
26    (@process [$($unwind_safe:tt)*] ($($var:ident $arg1:ident $arg2:ident $arg3:ident $arg4:ident)*) $($rest:tt)*) => {
27        closures!(@impl_for_args ($($var),*) FromWasmAbi [$($unwind_safe)*] $($var::from_abi($var) => $var $arg1 $arg2 $arg3 $arg4)*);
28        closures!(@process [$($unwind_safe)*] $($rest)*);
29    };
30
31    // Base case
32    (@process [$($unwind_safe:tt)*]) => {};
33
34    // A counter helper to count number of arguments.
35    (@count_one $ty:ty) => (1);
36
37    (@describe ( $($ty:ty),* )) => {
38        // Needs to be a constant so that interpreter doesn't crash on
39        // unsupported operations in debug mode.
40        const ARG_COUNT: u32 = 0 $(+ closures!(@count_one $ty))*;
41        inform(ARG_COUNT);
42        $(<$ty>::describe();)*
43    };
44
45    // This silly helper is because by default Rust infers `|var_with_ref_type| ...` closure
46    // as `impl Fn(&'outer_lifetime A)` instead of `impl for<'temp_lifetime> Fn(&'temp_lifetime A)`
47    // while `|var_with_ref_type: &A|` makes it use the higher-order generic as expected.
48    (@closure ($($ty:ty),*) $($var:ident)* $body:block) => (move |$($var: $ty),*| $body);
49
50    (@impl_for_fn $is_mut:literal [$($mut:ident)?] $Fn:ident $FnArgs:tt $FromWasmAbi:ident $($var_expr:expr => $var:ident $arg1:ident $arg2:ident $arg3:ident $arg4:ident)*) => (const _: () = {
51        impl<$($var,)* R> IntoWasmAbi for &'_ $($mut)? (dyn $Fn $FnArgs -> R + '_)
52        where
53            Self: WasmDescribe,
54        {
55            type Abi = WasmSlice;
56
57            fn into_abi(self) -> WasmSlice {
58                unsafe {
59                    let (a, b): (usize, usize) = mem::transmute(self);
60                    WasmSlice { ptr: a as u32, len: b as u32 }
61                }
62            }
63        }
64
65        // Generate invoke function that checks unwind_safe flag when unwinding is available
66        // unwind_safe flag is the MSV of the vtable pointer for a closure, distinguishing
67        // closures which are and are not unwind safe
68        #[cfg(all(feature = "std", target_arch = "wasm32", panic = "unwind"))]
69        #[allow(non_snake_case)]
70        unsafe extern "C-unwind" fn invoke<$($var: $FromWasmAbi,)* R: ReturnWasmAbi>(
71            a: usize,
72            b: usize,
73            $(
74            $arg1: <$var::Abi as WasmAbi>::Prim1,
75            $arg2: <$var::Abi as WasmAbi>::Prim2,
76            $arg3: <$var::Abi as WasmAbi>::Prim3,
77            $arg4: <$var::Abi as WasmAbi>::Prim4,
78            )*
79        ) -> WasmRet<R::Abi> {
80            if a == 0 {
81                throw_str("closure invoked recursively or after being dropped");
82            }
83            let unwind_safe = (b & 0x80000000) != 0;
84            let b = b & 0x7FFFFFFF;
85            let ret = {
86                let f: & $($mut)? dyn $Fn $FnArgs -> R = mem::transmute((a, b));
87                $(
88                    let $var = $var::Abi::join($arg1, $arg2, $arg3, $arg4);
89                )*
90                if unwind_safe {
91                    maybe_catch_unwind(AssertUnwindSafe(|| f($($var_expr),*)))
92                } else {
93                    f($($var_expr),*)
94                }
95            };
96            ret.return_abi().into()
97        }
98
99        // When unwinding is not available, generate a simple invoke function
100        #[cfg(not(all(feature = "std", target_arch = "wasm32", panic = "unwind")))]
101        #[allow(non_snake_case)]
102        unsafe extern "C-unwind" fn invoke<$($var: $FromWasmAbi,)* R: ReturnWasmAbi>(
103            a: usize,
104            b: usize,
105            $(
106            $arg1: <$var::Abi as WasmAbi>::Prim1,
107            $arg2: <$var::Abi as WasmAbi>::Prim2,
108            $arg3: <$var::Abi as WasmAbi>::Prim3,
109            $arg4: <$var::Abi as WasmAbi>::Prim4,
110            )*
111        ) -> WasmRet<R::Abi> {
112            if a == 0 {
113                throw_str("closure invoked recursively or after being dropped");
114            }
115            let ret = {
116                let f: & $($mut)? dyn $Fn $FnArgs -> R = mem::transmute((a, b));
117                $(
118                    let $var = $var::Abi::join($arg1, $arg2, $arg3, $arg4);
119                )*
120                f($($var_expr),*)
121            };
122            ret.return_abi().into()
123        }
124
125        #[allow(clippy::fn_to_numeric_cast)]
126        impl<$($var,)* R> WasmDescribe for dyn $Fn $FnArgs -> R + '_
127        where
128            $($var: $FromWasmAbi,)*
129            R: ReturnWasmAbi,
130        {
131            #[cfg_attr(wasm_bindgen_unstable_test_coverage, coverage(off))]
132            fn describe() {
133                inform(FUNCTION);
134                inform(invoke::<$($var,)* R> as *const () as usize as u32);
135                closures!(@describe $FnArgs);
136                R::describe();
137                R::describe();
138            }
139        }
140
141        unsafe impl<$($var,)* R> WasmClosure for dyn $Fn $FnArgs -> R + '_
142        where
143            Self: WasmDescribe,
144        {
145            const IS_MUT: bool = $is_mut;
146        }
147
148        impl<T, $($var,)* R> IntoWasmClosure<dyn $Fn $FnArgs -> R> for T
149        where
150            T: 'static + $Fn $FnArgs -> R,
151        {
152            fn unsize(self: Box<Self>) -> Box<dyn $Fn $FnArgs -> R> { self }
153        }
154    };);
155
156    (@impl_for_args $FnArgs:tt $FromWasmAbi:ident [$($maybe_unwind_safe:tt)*] $($var_expr:expr => $var:ident $arg1:ident $arg2:ident $arg3:ident $arg4:ident)*) => {
157        closures!(@impl_for_fn false [] Fn $FnArgs $FromWasmAbi $($var_expr => $var $arg1 $arg2 $arg3 $arg4)*);
158        closures!(@impl_for_fn true [mut] FnMut $FnArgs $FromWasmAbi $($var_expr => $var $arg1 $arg2 $arg3 $arg4)*);
159
160        // The memory safety here in these implementations below is a bit tricky. We
161        // want to be able to drop the `Closure` object from within the invocation of a
162        // `Closure` for cases like promises. That means that while it's running we
163        // might drop the `Closure`, but that shouldn't invalidate the environment yet.
164        //
165        // Instead what we do is to wrap closures in `Rc` variables. The main `Closure`
166        // has a strong reference count which keeps the trait object alive. Each
167        // invocation of a closure then *also* clones this and gets a new reference
168        // count. When the closure returns it will release the reference count.
169        //
170        // This means that if the main `Closure` is dropped while it's being invoked
171        // then destruction is deferred until execution returns. Otherwise it'll
172        // deallocate data immediately.
173
174        #[allow(non_snake_case, unused_parens)]
175        impl<T, $($var,)* R> WasmClosureFnOnce<dyn FnMut $FnArgs -> R, $FnArgs, R> for T
176        where
177            T: 'static + (FnOnce $FnArgs -> R),
178            $($var: $FromWasmAbi + 'static,)*
179            R: ReturnWasmAbi + 'static,
180            $($maybe_unwind_safe)*
181        {
182            fn into_fn_mut(self) -> Box<dyn FnMut $FnArgs -> R> {
183                let mut me = Some(self);
184                Box::new(move |$($var),*| {
185                    let me = me.take().expect_throw("FnOnce called more than once");
186                    me($($var),*)
187                })
188            }
189
190            fn into_js_function(self) -> JsValue {
191                use alloc::rc::Rc;
192                use crate::__rt::WasmRefCell;
193
194                let rc1 = Rc::new(WasmRefCell::new(None));
195                let rc2 = rc1.clone();
196
197                let closure = Closure::once(closures!(@closure $FnArgs $($var)* {
198                    let result = self($($var),*);
199
200                    // And then drop the `Rc` holding this function's `Closure`
201                    // alive.
202                    debug_assert_eq!(Rc::strong_count(&rc2), 1);
203                    let option_closure = rc2.borrow_mut().take();
204                    debug_assert!(option_closure.is_some());
205                    drop(option_closure);
206
207                    result
208                }));
209
210                let js_val = closure.as_ref().clone();
211
212                *rc1.borrow_mut() = Some(closure);
213                debug_assert_eq!(Rc::strong_count(&rc1), 2);
214                drop(rc1);
215
216                js_val
217            }
218        }
219
220        #[allow(non_snake_case, unused_parens)]
221        impl<T, $($var,)* R> WasmClosureFnOnceAbort<dyn FnMut $FnArgs -> R, $FnArgs, R> for T
222        where
223            T: 'static + (FnOnce $FnArgs -> R),
224            $($var: $FromWasmAbi + 'static,)*
225            R: ReturnWasmAbi + 'static,
226        {
227            fn into_fn_mut(self) -> Box<dyn FnMut $FnArgs -> R> {
228                let mut me = Some(self);
229                Box::new(move |$($var),*| {
230                    let me = me.take().expect_throw("FnOnce called more than once");
231                    me($($var),*)
232                })
233            }
234
235            fn into_js_function(self) -> JsValue {
236                use alloc::rc::Rc;
237                use crate::__rt::WasmRefCell;
238
239                let rc1 = Rc::new(WasmRefCell::new(None));
240                let rc2 = rc1.clone();
241
242                let closure = Closure::once_aborting(closures!(@closure $FnArgs $($var)* {
243                    let result = self($($var),*);
244
245                    // And then drop the `Rc` holding this function's `Closure`
246                    // alive.
247                    debug_assert_eq!(Rc::strong_count(&rc2), 1);
248                    let option_closure = rc2.borrow_mut().take();
249                    debug_assert!(option_closure.is_some());
250                    drop(option_closure);
251
252                    result
253                }));
254
255                let js_val = closure.as_ref().clone();
256
257                *rc1.borrow_mut() = Some(closure);
258                debug_assert_eq!(Rc::strong_count(&rc1), 2);
259                drop(rc1);
260
261                js_val
262            }
263        }
264    };
265
266    ([$($unwind_safe:tt)*] $( ($($var:ident $arg1:ident $arg2:ident $arg3:ident $arg4:ident)*) )*) => ($(
267        closures!(@impl_for_args ($($var),*) FromWasmAbi [$($maybe_unwind_safe)*] $($var::from_abi($var) => $var $arg1 $arg2 $arg3 $arg4)*);
268    )*);
269}
